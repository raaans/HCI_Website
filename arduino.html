Arduino workshops

In these workshops, we learn to program hardware (such as servo-motors, LED displays, etc) with the help of an Adruino Uno; a small board with a simple microcontroller on it.

Workshop 1

In the first workshop, there were 7 exercises. I'll show you four of the things I programmed in this challenge:

3C: Blinky blinky LEDs

First, we learned how to control LEDs: wether they were on or off and to control their brightness through digital pulse-width modulation. In this first example I programmed two LEDs, one of which blinks twice as fast as the other.

×
× int rLed = 10;
int gLed = 9;

void setup() {
  pinMode(rLed, OUTPUT);
  pinMode(gLed, OUTPUT);
}

void loop() {
  digitalWrite(rLed, HIGH);
  digitalWrite(gLed, HIGH);
  delay(500);

  digitalWrite(gLed, LOW);
  delay(500);

  digitalWrite(rLed, LOW);
  digitalWrite(gLed, HIGH);
  delay(500);

  digitalWrite(gLed, LOW);
  delay(500);
}

4C: Slow Fade

In this exercise, I made sure two LEDs slowly faded in and out. Where the green LED fades out, the red LED fades in, and vice-versa.

×
× int rLed = 10;
int gLed = 9;

void setup() {
 pinMode(rLed, OUTPUT);
 pinMode(gLed, OUTPUT);
}

void loop() {
 int x = 1;

 for (int i = 0; i > -1; i = i + x) {
  analogWrite(rLed, i);
  analogWrite(gLed, 255-i);

  if (i == 255) {
    x = -1;
  }
   delay(5);
 }
}

6C: This has potentio

Using a potentiometer, you can turn a knob to generate a number by measuring the resistance it has. Knob goes left, resistance goes down. Knob goes right, resistance goes up. You can of course use this to control the brightness of LEDs:

×
× int rLed = 10;
int gLed = 9;
float sensorValue = 0;
int sensorPin = A0;

void setup() {
 pinMode(rLed, OUTPUT);
 pinMode(gLed, OUTPUT);
 pinMode(sensorPin, INPUT);
 Serial.begin(9600);
}

void loop() {
 sensorValue = analogRead(sensorPin);
 Serial.println(sensorValue);
 sensorValue = (sensorValue/1023) * 255;
 analogWrite(rLed, sensorValue);
 analogWrite(gLed, 255-sensorValue);
}

10D: Hot Stuff

We've seen LEDs that turn off and on. We've seen LEDs that turn off and on, but slowly. We've even seen LEDs that are controlled by ambient light. But have you seen LEDs that are controlled by body heat? Don't think so!

In this exersize, I used a thermal resistor to make sure the LED is on when it reaches the middle point between the ambient temperature and my skin temperature.

×
× int led = 13;
int sense = A0;

void setup() {
 pinMode(led, OUTPUT);
 pinMode(sense, INPUT);
 Serial.begin(9600);
}

void loop() {
 int val = analogRead(sense);
 // read sensorpin
 Serial.println(val); // print to serial monitor
 delay(10); // wait

 if(val > 538) {
  analogWrite(led, 255);
 }
 else {
  analogWrite(led, 0);
 }
}

11D: Stay away

The kit came with a radar, so we can't let that go unused. In this exercise, we use the arduino and the radar to measure the distance between the arduino and whatever is in front of the radar.

If there is an object or person less than 140cm away, the red light comes on. If the object is between 140 and 160cm away, the yellow light is on instead. Otherwise, the green light is there to tell you all's well.

×
× const int echoPin = 6;
const int trigPin = 7;
int rLed = 9;
int yLed = 5;
int gLed = 3;

void setup() {
 pinMode(echoPin, INPUT);
 pinMode(trigPin, OUTPUT);
 pinMode(rLed, OUTPUT);
 pinMode(yLed, OUTPUT);
 pinMode(gLed, OUTPUT);
 Serial.begin(9600);
}

void loop() {
 // send a pulse
 digitalWrite(trigPin, LOW);
 delayMicroseconds(2);
 digitalWrite(trigPin, HIGH);
 delayMicroseconds(5);
 digitalWrite(trigPin, LOW);
 // wait for the echo
 long duration = pulseIn(echoPin, HIGH);

 int cm = duration / 29 / 2;
 Serial.print(cm);
 Serial.println();
 delay(100);

 //if the measured distance is lower than 50 cm, turn on LED
 if(cm < 140) {
  analogWrite(rLed, 255);
  analogWrite(yLed, 0);
  analogWrite(gLed, 0);
 }

 else if (cm < 170 && (cm > 140)) {
  analogWrite(rLed, 0);
  analogWrite(yLed, 255);
  analogWrite(gLed, 0);
 }

 else {
  analogWrite(rLed, 0);
  analogWrite(yLed, 0);
  analogWrite(gLed, 255);
 }

 delay(100);
}
